    // Particle cursor trail

    const canvas = document.getElementById('particles');

    const ctx = canvas.getContext('2d');

    let W, H; const DPR = Math.min(2, window.devicePixelRatio || 1);

    function resize(){ W = canvas.width = innerWidth * DPR; H = canvas.height = innerHeight * DPR; canvas.style.width = innerWidth + 'px'; canvas.style.height = innerHeight + 'px'; ctx.scale(DPR, DPR); }

    resize(); addEventListener('resize', resize);


    const parts = [];

    const hueBase = 355; // red

    function spawn(x,y){

      for(let i=0;i<3;i++){

        parts.push({ x, y, vx:(Math.random()-0.5)*2.2, vy:(Math.random()-0.5)*2.2, life: 36+Math.random()*18 });

      }

    }

    addEventListener('pointermove', e=>{ const r = canvas.getBoundingClientRect(); spawn(e.clientX - r.left, e.clientY - r.top); });


    function tick(){

      ctx.clearRect(0,0,innerWidth,innerHeight);

      for(let i=parts.length-1; i>=0; i--){

        const p = parts[i]; p.life-=1; if(p.life<=0){ parts.splice(i,1); continue; }

        p.x += p.vx; p.y += p.vy; p.vy += 0.02;

        const alpha = Math.max(0, p.life/50);

        const glow = `hsla(${hueBase}, 95%, 60%, ${alpha})`;

        ctx.beginPath(); ctx.arc(p.x, p.y, 2, 0, Math.PI*2);

        ctx.fillStyle = glow; ctx.shadowBlur = 12; ctx.shadowColor = glow; ctx.fill();

      }

      requestAnimationFrame(tick);

    }

    tick();


    // Tilt micro-interactions for cards/buttons

    function addTilt(sel){

      document.querySelectorAll(sel).forEach(el=>{

        el.addEventListener('pointermove', (e)=>{

          const r = el.getBoundingClientRect();

          const dx = (e.clientX - r.left)/r.width - .5;

          const dy = (e.clientY - r.top)/r.height - .5;

          el.style.transform = `rotateX(${(-dy*6).toFixed(2)}deg) rotateY(${(dx*6).toFixed(2)}deg) translateY(-6px)`;

        });

        el.addEventListener('pointerleave', ()=>{ el.style.transform = ''; });

      });

    }

    addTilt('.card'); addTilt('.btn');
